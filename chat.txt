            if (quantity > products[i].stock)

            cart[cartItemCount].productCode = products[i].code;
            strcpy(cart[cartItemCount].productName, products[i].name);

            cartItemCount++;

            products[i].stock = products[i].stock - quantity;

            printf("\nItem added to cart successfully.\n");

                revenuePerProduct[j] += cart[i].quantity * cart[i].price;
            }
        }
    }

    totalTransactions++;
}

    printf("----------------------------------------\n");

    for (i = 0; i < productCount; i++)
    {
        printf("%-12s\t%d\t\t%.2f\n",
               products[i].name,
               itemsSold[i],
               revenuePerProduct[i]);
    }

    printf("----------------------------------------\n");
}

/* Reset sales report */
void resetSalesReport()
{
    int i;
    for (i = 0; i < MAX_PRODUCTS; i++)
    {
        itemsSold[i] = 0;
        revenuePerProduct[i] = 0.0;

    totalTransactions = 0;

}
